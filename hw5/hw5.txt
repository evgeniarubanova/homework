# Мне было не очень понятно, в каком порядке нужно выполнять задание.
# Я выполнила его след. образом:
# 1)сначала удалила каждую третью буква, а затем перевернула слово.
# 2)нумеровала буквы в слове от 0. (так в слове "подарок" третьей буквой для меня была буква "а")
arr=[] 
while True:
	# Ввод слова во временную строку
	a=input('Введите слово ')
	# Если временная строка пуста, то цикл ввода заканчивается
	if a=='':
		break	
	# Добавление строки в список(массив) 
	arr.append(''.join(a[i+1:i+3] for i in range(0, len(a), 3))[::-1])

for w in arr[::1]:
     print(w)
